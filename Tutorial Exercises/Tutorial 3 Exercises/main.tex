def TreeInsert(root, key):
    if root is None:
        return Node(key, goLeft=True)
    if key == root.key:
        if root.goLeft:
            root.left = TreeInsert(root.left, key)
        else:
            root.right = TreeInsert(root.right, key)
        root.goLeft = not root.goLeft
    elif key < root.key:
        root.left = TreeInsert(root.left, key)
    else:
        root.right = TreeInsert(root.right, key)
    return root
